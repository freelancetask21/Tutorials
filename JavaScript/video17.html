                let Getmygrade = function(currentmarks,totalmarks){
                    let  percentage = currentmarks/totalmarks*100;
                 
                   let mygrade = ' '
                     if(percentage>=90){
                         mygrade = 'A';
                     }
                 
                     else if(percentage>=80)
                     {
                         mygrade = 'B';
                     }
                     else if(percentage>=70)
                     {
                         mygrade = 'C';
                     }
                     else if(percentage>=60)
                     {
                         mygrade = 'D';
                     }
                     else
                     {
                        mygrade ='Fail'
                 
                     }
                     return ` your Grade is: ${mygrade} and your percentage is: ${percentage}`
                 }
                 let yourResult =  Getmygrade(98,100);
                 console.log(yourResult);
                 //  your Grade is: Fail and your percentage is: 34 
                 // your Grade is: A and your percentage is: 98
